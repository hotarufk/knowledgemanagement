SELECT case `status` when '1' then 'Pre-Register' when '2' then 'In Progress' when '3' then 'Closed-Cancelled' when '4' then 'Closed-Pending' when '5' then 'Closed-Finished' else 'Unon' end 'Status',COUNT(CASE when`month_of_register` ='07.2014' and `end_date` >='2014-07-01' THEN `id` ELSE NULL END) AS this_month, COUNT(CASE when`month_of_register` <'07.2014' and `end_date` >='2014-07-01' THEN `id` ELSE NULL END) AS overmonth FROM tbl_data group by `status`
